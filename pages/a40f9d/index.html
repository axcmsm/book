    @Autowired
    private StringRedisTemplate stringRedisTemplate;//key value 都是字符串
@Test
    void TestOpsFor(){
        //操作string
        stringRedisTemplate.opsForValue().set("name","张三",3, TimeUnit.SECONDS);
        stringRedisTemplate.opsForValue().append("name",",他是好人！");
        String name = stringRedisTemplate.opsForValue().get("name");
        System.out.println(name);
        //操作list
        stringRedisTemplate.opsForList().leftPushAll("names","张三","李四");
        ArrayList<String> arrayList = new ArrayList<>();
        arrayList.add("张珊na");
        stringRedisTemplate.opsForList().leftPushAll("nams1",arrayList);
        stringRedisTemplate.opsForList().range("names",0,-1);//查看
        //set
        stringRedisTemplate.opsForSet().add("set1","张撒比","张三");
        stringRedisTemplate.opsForSet().members("set1");//查看
        //排序set
        stringRedisTemplate.opsForZSet().add("zsets","张三",1);//放入元素可排序值
        stringRedisTemplate.opsForZSet().range("zsets",0,-1);//范围查询
        stringRedisTemplate.opsForZSet().rangeByScoreWithScores("zsets",0,1000);//获取分数排序 分数范围 显示分数
        //hash
        stringRedisTemplate.opsForHash().put("map","names","张三");
        stringRedisTemplate.opsForHash().get("map","names");
        stringRedisTemplate.opsForHash().values("map");
        HashMap<String, String> map = new HashMap<>();
        map.put("name","张三");
        map.put("age","15");
        stringRedisTemplate.opsForHash().putAll("map",map);
    }

        //高亮查询加封装
        SearchResponse search = restHighLevelClient.search(
                new SearchRequest("ems").source(
                        new SearchSourceBuilder()
                        .query(QueryBuilders.termQuery("name","张"))
                        .from(0)//起始页=(当前页-1)*size
                        .size(20)
                        .sort("age",SortOrder.DESC)
                        .highlighter(new HighlightBuilder().field("*").requireFieldMatch(false))
                )
                , RequestOptions.DEFAULT);

        //查询
        ArrayList<Emp> objects = new ArrayList<>();
        Arrays.stream(search.getHits().getHits()).forEach(x->{
            Map<String, Object> sourceAsMap = x.getSourceAsMap();
            Emp emp = new Emp();
            emp.setId(x.getId());
            emp.setAge(Integer.parseInt(sourceAsMap.get("age").toString()));
            // emp.setBir(new Date(sourceAsMap.get("bir").toString()));
            emp.setName(sourceAsMap.get("name").toString());
            emp.setContent(sourceAsMap.get("content").toString());
            emp.setAddress(sourceAsMap.get("address").toString());
            //替换高亮
            Map<String, HighlightField> fields = x.getHighlightFields();
            if (fields.containsKey("name")) emp.setName(fields.get("name").fragments()[0].toString());//替换高亮的字段
            if (fields.containsKey("content")) emp.setName(fields.get("content").fragments()[0].toString());//替换高亮的字段
            objects.add(emp);
        });
        objects.forEach(System.out::println);
    }

/**
 * 使用在类上 将emp对象映射成Es中的一条json文档
 *   indexName: 用来指定这个对象转为json文档存入那个索引中,且不能存在该索引(自动会创建)
 * 在属性上标注：Field作为字段类型
 * 当springboot启动当放文档就会创建
 */
@Data
@NoArgsConstructor
@AllArgsConstructor
@Document(indexName = "ems")//type 是类型名 现此类型被慢慢弃用
public class Emp {

    @Id //将对象中id属性和文档中_id 一 一对应
    private String id;
    //用在属性上 mapping中的属性，type:指定类型,analyzer:指定分词器
    @Field(type = FieldType.Text,analyzer = "ik_max_word")//只有test才能分词
    private String name;
    @Field(type = FieldType.Integer)
    private Integer age;
//    @Field(type = FieldType.Date,format = DateFormat.date_time)
    @Field(type = FieldType.Date)
    @JsonFormat(pattern = "yyyy-MM-dd")
    private Date bir;
    @Field(type = FieldType.Text,analyzer = "ik_max_word")//只有test才能分词
    private String content;
    @Field(type = FieldType.Text,analyzer = "ik_max_word")//只有test才能分词
    private String address;
}
