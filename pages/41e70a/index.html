# 文件夹拷贝(多进程)

import multiprocessing
import os 
# file_name ⽂件名 
# source_dir 源⽂件⽬录
# dest_dir ⽬标⽂件⽬录 
def copy_work(file_name, source_dir, dest_dir): 
    # 拼接路径 
    source_path = source_dir+"/"+file_name 
    dest_path = dest_dir+"/"+file_name 
    print(source_path, "----->", dest_path) 
    # 打开源⽂件、创建⽬标⽂件 
    with open(source_path,"rb") as source_file: 
        with open(dest_path,"wb") as dest_file: 
            while True: 
                # 循环读取数据 
                file_data = source_file.read(1024) 
                if file_data:
                    # 循环写⼊到⽬标⽂件 
                    dest_file.write(file_data) 
                else:
                    break
if __name__ == '__main__':
    # 1、定义源⽂件⽬录和⽬标⽂件夹的⽬录
    source_dir = "test"
    dest_dir = "/home/teahcer/桌⾯/test" 
    try:
        # 2、创建⽬标⽂件夹⽬录 
        os.mkdir(dest_dir)
    except: 
        print("⽬标⽂件夹已经存在，未创建~") 
    # 3、列表得到所有的源⽂件中的⽂件
    file_list = os.listdir(source_dir)
    print(file_list)
    # 4、创建进程池
    pool = multiprocessing.Pool(3) 
    # 5、for 循环，依次拷⻉每个⽂件
    for file_name in file_list:
        # copy_work(file_name, source_dir, dest_dir) 
        pool.apply_async(copy_work, args=(file_name, source_dir, dest_dir)) 
    # 6、关闭进程池
    pool.close() 
	# 7、设置主进程等待⼦进程执⾏结束再退出 
	pool.join()

# 查看微信撤回消息

1.Win+R输入cmd进入到CMD窗口下，执行 python -m pip install -U pip setuptools 进行升级
2.输入 python -m pip install matplotlib 进行自动的安装，系统会自动下载安装包
3.进入pycharm 创建新的.py文件
4.粘贴下面代码执行测试
import matplotlib.pyplot as plt
labels='frogs','hogs','dogs','logs'
sizes=15,20,45,10
colors='yellowgreen','gold','lightskyblue','lightcoral'
explode=0,0.1,0,0
plt.pie(sizes,explode=explode,labels=labels,colors=colors,autopct='%1.1f%%',shadow=True,startangle=50)
plt.axis('equal')
plt.show()
