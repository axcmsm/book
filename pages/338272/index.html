# 分类功能

**查询接口 **Service

@Override
    public List<CategoryEntity> listWithTree() {
        //1. 查出所有的分类
        List<CategoryEntity> entities = baseMapper.selectList(null);

        //2. 组装成父子结构

            //2.1 找到所有一级分类
            List<CategoryEntity> oneList = entities.stream().filter((item) -> {
                return item.getParentCid() == 0; //父id等于0
            }).map((menu)->{
              //2.2根据父id查询子菜单进行封装
                menu.setChildren(getChildrens(menu,entities));
                return menu;
            })
             .sorted((item1,item2)->{ //排序 升序  排除出现空值的状况
                return (item1.getSort()==null?0:item1.getSort())-(item2.getSort()==null?0:item2.getSort());
            }).collect(Collectors.toList());//收集成list 返回




        return oneList;
    }
    //在集合中查询所有子菜单  递归查找
    private List<CategoryEntity> getChildrens(CategoryEntity root,List<CategoryEntity> all){

        List<CategoryEntity> children = all.stream().filter((item) -> {//过滤出所有父id 等于 root的id =>子菜单
            return item.getParentCid() == root.getCatId();
        }).map(categoryEntity -> {
            //再查询子菜单的子菜单
            categoryEntity.setChildren(getChildrens(categoryEntity,all));
            return categoryEntity;
        }).sorted((item1,item2)->{//排序 升序  排除出现空值的状况
            return (item1.getSort()==null?0:item1.getSort())-(item2.getSort()==null?0:item2.getSort());
        }).collect(Collectors.toList());

        return children;
    }
// 添加字段
	//字段不为空才返回字段
	@JsonInclude(JsonInclude.Include.NON_EMPTY)
	@TableField(exist=false)//子菜单 数据库字段不存在
	private List<CategoryEntity> children;

          byte[] binaryData = postPolicy.getBytes("utf-8");

    /**
 * 自定义校验注解
 */
@Target({ElementType.METHOD, ElementType.FIELD, ElementType.ANNOTATION_TYPE, ElementType.CONSTRUCTOR, ElementType.PARAMETER, ElementType.TYPE_USE})
@Retention(RUNTIME)
@Documented
@Constraint(
		//绑定
        validatedBy = {ListValueConstraintValidator.class}
)
public @interface ListValue {
    String message() default "{com.axmcmsm.common.valid.ListValue.message}";

    Class<?>[] groups() default {};

    Class<? extends Payload>[] payload() default {};

    //自定义值
    int[] vals();

}
/**
 * 自定义校验器
 */
public class ListValueConstraintValidator implements ConstraintValidator<ListValue,Integer> {

    private Set<Integer> set=new HashSet<>();

    //初始化方法
    @Override
    public void initialize(ListValue constraintAnnotation) {
        int[] vals = constraintAnnotation.vals();
        for (int val : vals) {
            //这里需要添加非空判断
            set.add(val);
        }
    }

    //是否校验成功

    /**
     *
     * @param value 需要校验的值
     * @param constraintValidatorContext  上下文环境消息
     * @return
     */
    @Override
    public boolean isValid(Integer value, ConstraintValidatorContext constraintValidatorContext) {
        return set.contains(value);
    }
}
