    hits(索引、文档、总数、匹配分数.....)

    #全表查询
GET /axcmsm/_doc/_search
#条件查询
GET /axcmsm/_doc/_search?q=name:老八 #eq 
#模糊查询+指定name、tags字段 
GET /axcmsm/_doc/_search
{
  "query":{
    "match":{
      "name":"老八"
    }
  },
  "_source":["name","tags"]
}
#排序通过age进行倒序排序
GET /axcmsm/_doc/_search
{
  "query":{
    "match":{
      "name":"老八"
    }
  },
  "sort":[{
    "age":{
      "order":"desc"
    }
  }]
}
#分页from起始下标，size取多少条 
GET /axcmsm/_doc/_search
{
  "query":{
    "match":{
      "name":"老八"
    }
  },
  "from":1,
  "size":2
} 
#多条件精确查询 用空格隔开即可
#must(and)、should(or)、must_not(not)
GET /axcmsm/_doc/_search
{
  "query":{
    "bool":{
      "must":[
        {
          "match":{"name":"老八"}
        },
        {
           "match":{"age":15}
        }
      ]
    }
  }
}
#过滤where(filter) gt、lt、eq、gte、lte
GET /axcmsm/_doc/_search
{
  "query":{
    "bool":{
        "must":[
          {
            "match":{"name":"老八"}
          }
        ],
        "filter":{
          "range":{
            "age":{
              "gt":20,
               "lte":200
            }
        }
      }
    }
  }
}
#高亮查询 pre_tags、post_tags是自定义
GET /axcmsm/_doc/_search
{
  "query":{
       "match":{"name":"老八"}
        },
        
        "highlight":{ 
          "pre_tags":"<p class='key' style='color:red'>",
           "post_tags":"</p>",
          "fields":{ "name":{}}
  }
}

    //todo 添加文档
    @Test
    void  testAddDocument() throws IOException {
        User user = new User();//创建对象
        user.setName("占山");

        //设置规则 put axcmsm_index/_doc/1

    //todo 获取文档 判断文档是否存在

    //todo 批量插入数据 项目使用
    @Test
    void  testBulkDocument() throws IOException {
        BulkRequest bulkRequest = new BulkRequest();

     		//todo 封装高亮的字段
        HighlightBuilder highlightBuilder = new HighlightBuilder();
        highlightBuilder.field("title");
        highlightBuilder.requireFieldMatch(false);//关闭多个字段高亮
        highlightBuilder.preTags("<span style='color:red'>");
        highlightBuilder.postTags("</span>");
        builder.highlighter(highlightBuilder);//设置高亮
		 //todo 执行搜索
        searchRequest.source(builder);
        SearchResponse search = restHighLevelClient.search(searchRequest, RequestOptions.DEFAULT);
        //todo 封装解析结果
        HashMap<String, Object> mapdata = new HashMap<>();
        ArrayList<Map<String, Object>> list = new ArrayList<>();
        for (SearchHit hit : search.getHits().getHits()) {
            //todo 解析高亮的字段 将高亮字段替换原来的字段
            Map<String, HighlightField> fields = hit.getHighlightFields();
            HighlightField title = fields.get("title");
            Map<String, Object> sourceAsMap = hit.getSourceAsMap();
            if (title!=null){//如果字段不为空
                Text[] fragments = title.fragments();
                String new_title="";
                for (Text text : fragments) {//拼接成新title
                    new_title+=text;
                }
                sourceAsMap.put("title",new_title);
            }
            //返回解析的高亮
            list.add(sourceAsMap);
        }
