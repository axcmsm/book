/**
     * 数据的导出
     * @param response
     * @throws Exception
     */
    @GetMapping("/api/testpoi")
    public void poitest(HttpServletResponse response) throws Exception {
//        XSSFWorkbook workbook = new XSSFWorkbook();//工作簿
        SXSSFWorkbook workbook = new SXSSFWorkbook();//工作簿 可支持百万数据导出 样式会减少不支持模板,替换了解析方式
        SXSSFSheet sheet = workbook.createSheet("测试");//工作表
        String[] titles = {"city", "hot"};
        SXSSFRow titleRow = sheet.createRow(0);
        Cell cell=null;
        for (int i = 0; i < 2; i++) {
            cell=titleRow.createCell(i);
            cell.setCellValue(titles[i]);
        }
        List<DataVo> query = jdbcTemplate.query("select CITYHOTTOP20.\"city\" as name,CITYHOTTOP20.\"hot\" as value from TEST.CITYHOTTOP20", new BeanPropertyRowMapper<DataVo>(DataVo.class));
       int rowIndex=1;
       Row row=null;
        for (DataVo dataVo : query) {
            row=sheet.createRow(rowIndex);
            cell = row.createCell(0);
            cell.setCellValue(dataVo.getName()) ;
            cell = row.createCell(1);
            cell.setCellValue(dataVo.getValue()) ;
            rowIndex++;
        }

        //一个流俩个头
        String filename="报表数据.xlsx";
        response.setHeader("content-disposition","attachment;filename="+new String(filename.getBytes(),"ISO-8859-1"));
        response.setContentType("application/vnd.opnxmlformats-officedocument.spreadsheetml.sheet");
        workbook.write(response.getOutputStream());
    }

    /**
     * 数据的导出 csv
     * @param response
     * @throws Exception
     */
    @GetMapping("/api/downloadCSV")
    public void downloadCSV(HttpServletResponse response) throws Exception {
        ServletOutputStream outputStream = response.getOutputStream();
        String filename="报表数据.csv";
        response.setHeader("content-disposition","attachment;filename="+new String(filename.getBytes(),"ISO-8859-1"));
        response.setContentType("text/csv");
        CSVWriter csvWriter = new CSVWriter(new OutputStreamWriter(outputStream, "utf-8"));
        List<DataVo> query = jdbcTemplate.query("select CITYHOTTOP20.\"city\" as name,CITYHOTTOP20.\"hot\" as value from TEST.CITYHOTTOP20", new BeanPropertyRowMapper<DataVo>(DataVo.class));
//        csvWriter.writeNext(); //一行
//        csvWriter.writeAll(); //全部
        String[] titles = {"city", "hot"};
        csvWriter.writeNext(titles);//写入头
        for (DataVo dataVo : query) {
            csvWriter.writeNext(new String[]{dataVo.getName(),dataVo.getValue()});
        }
        csvWriter.flush();
        csvWriter.close();
    }
